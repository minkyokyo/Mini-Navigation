cmake_minimum_required(VERSION 3.21)
project(my_navi LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(glfw3 CONFIG REQUIRED)
find_package(glad CONFIG REQUIRED)
find_package(glm CONFIG REQUIRED)

add_executable(my_navi
  src/main.cpp

    src/gfx/Mesh.cpp

    src/geometry/CubeGeometry.cpp
    src/geometry/BoxGeometry.cpp
    src/geometry/SphereGeometry.cpp
    src/geometry/PyramidGeometry.cpp
)

target_link_libraries(my_navi PRIVATE glfw glad::glad glm::glm)

target_include_directories(my_navi PRIVATE
    ${CMAKE_SOURCE_DIR}/include
)
